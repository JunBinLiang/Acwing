class Solution{
    public void solution(PrintWriter out,int A[][]){
        int n=A.length;int m=A[0].length;
        int p[][]=new int[n][m];
        int res=0;
        for(int i=0;i<m;i++){
            int sum=0;
            for(int j=0;j<n;j++){
                if(A[j][i]==1)sum++;
                else sum=0;
                p[j][i]=sum;
            }
        }


        for(int i=0;i<n;i++){
            int a[]=p[i];
            res=Math.max(res,solve(a));
        }






        out.println(res*3);
    }

    public int solve(int A[]){
        int res=0;
        Stack<int[]>stack=new Stack<>();
        for(int i=0;i<A.length;i++){
            if(stack.size()==0){
                stack.push(new int[]{A[i],1});
            }
            else{
                int pair[]=new int[]{A[i],1};
                int cnt=0;
                while(stack.size()>0&&pair[0]<=stack.peek()[0]){
                    int top[]=stack.pop();
                    cnt+=top[1];
                    res=Math.max(res,cnt*top[0]);
                }
                pair[1]+=cnt;
                stack.push(pair);
            }
        }
        int cnt=0;
        while(stack.size()>0){
            int top[]=stack.pop();
            cnt+=top[1];
            res=Math.max(res,cnt*top[0]);
        }
        return res;

    }





}
