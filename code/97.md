class Solution{
    int mod=9901;
    public void solution(PrintWriter out,int A,int B){
        if(A==0){
            out.println(0);
            return;
        }
        Map<Integer,Integer>f=new HashMap<>();
        for(int i=2;i<=A;i++){
            if(i>A||A==1)break;
            int cnt=0;
            while(A%i==0){
                A/=i;cnt++;
            }
            if(cnt>0){
                put(f,(i%mod),cnt);
            }
        }
        if(A!=1){
            put(f,(A%mod),1);
        }

        long res=1;
        for(Integer p:f.keySet()){
            res*=(sum(p,f.get(p)*B));
            res%=mod;
        }
        out.println(res);
    }

    public int sum(int p,int k){//p^0+p^1+...+p^k
        if(k==0)return 1;

        int mid=k/2;
        if(k%2==1){
            int q=quick(p,mid+1);
            return ((q+1)*sum(p,k/2))%mod;
        }
        else{
            int q1=quick(p,mid);
            int q2=quick(p,mid+1);
            return (((q2+1)*sum(p,k/2-1))%mod+q1)%mod;
        }
    }

    public void put(Map<Integer,Integer>f,int key,int cnt){
        if(!f.containsKey(key))f.put(key,cnt);
        else f.put(key,f.get(key)+cnt);
    }

    public int quick(int a,int b){
        int res=1;int pow=a;
        for(int i=0;i<32;i++){
            if((b&(1<<i))!=0){
                res*=pow;
                res%=mod;
            }
            pow*=pow;
            pow%=mod;
        }
        return res;
    }



}
