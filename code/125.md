class Sol{
    public void solution(PrintWriter out,int A[][]){
        long res=Long.MIN_VALUE;
        long sum=0;

        Arrays.sort(A,(a,b)->{
            return (b[0]+b[1])-(a[0]+a[1]);
        });

        for(int i=A.length-1;i>=0;i--){
            res=Math.max(res,sum-A[i][1]);
            sum+=A[i][0];
        }
