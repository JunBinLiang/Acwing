class Sol{
    long dp[][];
    public void solution(PrintWriter out,int A[][]){
        int n=A.length;
        dp=new long[(1<<n)+10][n+1];
        for(int i=0;i<dp.length;i++){
            Arrays.fill(dp[i],-1);
        }

        int state=(1<<n)-1;
        state--;

        long res=dfs(A,state,0);
        out.println(res);
    }

    public long dfs(int A[][],int state,int v){
        if(state==0){
            if(v!=A.length-1)return 100000000;
            return 0;
        }
        if(dp[state][v]!=-1)return dp[state][v];
        long res=Integer.MAX_VALUE;
        for(int u:A[v]){
            if((state&(1<<u))!=0){
                res=Math.min(res,A[v][u]+dfs(A,state^(1<<u),u));
            }
        }
        dp[state][v]=res;
        return res;
    }





}
